# react-live-search

## Install

```bash
npm install --save react-live-search
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-live-search'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
