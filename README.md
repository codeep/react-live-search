# react-search-input

## Install

```bash
npm install --save react-search-input
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-search-input'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
